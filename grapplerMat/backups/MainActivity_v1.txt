class MainActivity : ComponentActivity() {
    private val sppUuid: UUID = UUID.fromString("00001101-0000-1000-8000-00805F9B34FB")

    private var socket: BluetoothSocket? = null
    private var output: OutputStream? = null
    private var input: InputStream? = null
    private var readerThread: Thread? = null
    private var onBluetoothMessage: ((String) -> Unit)? = null

    private val permissionLauncher = registerForActivityResult(
        ActivityResultContracts.RequestMultiplePermissions()
    ) {}

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        requestBluetoothPermissions()

        setContent {
            GrapplerTheme {
                GrapplerScreen(
                    getDevices = { getPairedDevices() },
                    connectDevice = { connectToDevice(it) },
                    sendCommand = { sendCommand(it) },
                    disconnect = { disconnect() },
                    setMessageListener = { listener -> onBluetoothMessage = listener }
                )
            }
        }
    }

    private fun requestBluetoothPermissions() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {
            permissionLauncher.launch(
                arrayOf(
                    Manifest.permission.BLUETOOTH_CONNECT,
                    Manifest.permission.BLUETOOTH_SCAN
                )
            )
        }
    }

    @SuppressLint("MissingPermission")
    private fun getPairedDevices(): List<BluetoothDevice> {
        val adapter = BluetoothAdapter.getDefaultAdapter() ?: return emptyList()

        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {
            val ok = ContextCompat.checkSelfPermission(
                this,
                Manifest.permission.BLUETOOTH_CONNECT
            ) == PackageManager.PERMISSION_GRANTED
            if (!ok) return emptyList()
        }

        return adapter.bondedDevices.toList().sortedBy { it.name ?: "" }
    }

    @SuppressLint("MissingPermission")
    private suspend fun connectToDevice(device: BluetoothDevice): Boolean = withContext(Dispatchers.IO) {
        try {
            disconnect()
            BluetoothAdapter.getDefaultAdapter()?.cancelDiscovery()

            val s = device.createRfcommSocketToServiceRecord(sppUuid)
            s.connect()

            socket = s
            output = s.outputStream
            input = s.inputStream

            startReaderThread()
            true
        } catch (_: Exception) {
            false
        }
    }

    private fun sendCommand(cmd: String): Boolean {
        return try {
            val outStream = output ?: return false
            val s = socket ?: return false
            if (!s.isConnected) return false

            // Za dugmad šaljemo tačno jedan karakter: g, f, h, o, c, 1, 2, ?
            // Za slider komande već šaljemo newline: A090\n, B120\n
            outStream.write(cmd.toByteArray(Charsets.US_ASCII))
            outStream.flush()
            true
        } catch (_: Exception) {
            false
        }
    }

    private fun startReaderThread() {
        readerThread = Thread {
            val buffer = StringBuilder()
            try {
                while (socket?.isConnected == true) {
                    val b = input?.read() ?: break
                    if (b == '\n'.code) {
                        val line = buffer.toString().trim()
                        buffer.clear()
                        if (line.isNotEmpty()) {
                            runOnUiThread { onBluetoothMessage?.invoke(line) }
                        }
                    } else if (b != '\r'.code) {
                        buffer.append(b.toChar())
                    }
                }
            } catch (_: Exception) {
            }
        }
        readerThread?.start()
    }

    private fun disconnect() {
        try { input?.close() } catch (_: Exception) {}
        try { output?.close() } catch (_: Exception) {}
        try { socket?.close() } catch (_: Exception) {}
        input = null
        output = null
        socket = null
        readerThread = null
    }

    override fun onDestroy() {
        disconnect()
        super.onDestroy()
    }
}

@Composable
fun GrapplerScreen(
    getDevices: () -> List<BluetoothDevice>,
    connectDevice: suspend (BluetoothDevice) -> Boolean,
    sendCommand: (String) -> Boolean,
    disconnect: () -> Unit,
    setMessageListener: (((String) -> Unit)?) -> Unit
) {
    val scope = rememberCoroutineScope()
    val scroll = rememberScrollState()

    var devices by remember { mutableStateOf<List<BluetoothDevice>>(emptyList()) }
    var selectedDevice by remember { mutableStateOf<BluetoothDevice?>(null) }
    var connected by remember { mutableStateOf(false) }
    var status by remember { mutableStateOf("OFFLINE") }
    var servo1 by remember { mutableStateOf(SERVO1_HOME) }
    var servo2 by remember { mutableStateOf(SERVO2_HOME) }
    var logs by remember { mutableStateOf(listOf<String>()) }
    var lastCommand by remember { mutableStateOf("Nema") }

    fun now(): String = SimpleDateFormat("HH:mm:ss", Locale.getDefault()).format(Date())

    fun log(msg: String) {
        logs = (logs + "[${now()}] $msg").takeLast(80)
    }

    fun send(cmd: String, name: String): Boolean {
        if (!connected) {
            log("WARN: sistem nije povezan")
            return false
        }

        val ok = sendCommand(cmd)

        if (ok) {
            lastCommand = name
            log("TX: $name → '${cmd.trim()}'")
        } else {
            log("ERROR: komanda nije poslana")
        }

        return ok
    }

    LaunchedEffect(Unit) {
        devices = getDevices()
        log("Aplikacija pokrenuta")
        setMessageListener { line -> log("RX: $line") }
    }

    Surface(color = BG, modifier = Modifier.fillMaxSize()) {
        Column(
            modifier = Modifier
                .fillMaxSize()
                .verticalScroll(scroll)
                .padding(14.dp)
        ) {
            HeaderCard(status = status, connected = connected, deviceName = selectedDevice?.name ?: "—")
            Spacer(Modifier.height(12.dp))

            ConnectionCard(
                devices = devices,
                selectedDevice = selectedDevice,
                status = status,
                onRefresh = {
                    devices = getDevices()
                    log("Lista uređaja osvježena")
                },
                onSelect = { selectedDevice = it },
                onConnect = {
                    val d = selectedDevice
                    if (d == null) {
                        log("WARN: nije izabran uređaj")
                    } else {
                        scope.launch {
                            status = "CONNECTING"
                            log("Povezivanje na ${d.name ?: d.address}...")
                            val ok = connectDevice(d)
                            connected = ok
                            status = if (ok) "ONLINE" else "OFFLINE"
                            if (ok) {
                                lastCommand = "Home"
                                log("OK: povezano na ${d.name ?: d.address}")
                                sendCommand("h")
                            } else {
                                log("ERROR: konekcija nije uspjela")
                            }
                        }
                    }
                },
                onDisconnect = {
                    disconnect()
                    connected = false
                    status = "OFFLINE"
                    log("Konekcija prekinuta")
                }
            )
            Spacer(Modifier.height(12.dp))

            MetricsGrid(servo1 = servo1, servo2 = servo2, grip = gripIntensity(servo1, servo2), lastCommand = lastCommand)
            Spacer(Modifier.height(12.dp))

            ControlPanel(
                onGrab = {
                    if (send("g", "UHVATI")) {
                        servo1 = SERVO1_PULL
                        servo2 = SERVO2_PULL
                    }
                },
                onStop = {
                    if (send("f", "E-STOP")) {
                        servo1 = SERVO1_HOME
                        servo2 = SERVO2_HOME
                    }
                },
                onHome = {
                    if (send("h", "HOME")) {
                        servo1 = SERVO1_HOME
                        servo2 = SERVO2_HOME
                    }
                },
                onOpen = {
                    if (send("o", "OTVORI")) {
                        servo1 = SERVO1_HOME
                        servo2 = SERVO2_HOME
                    }
                },
                onClose = {
                    if (send("c", "ZATEGNI")) {
                        servo1 = SERVO1_PULL
                        servo2 = SERVO2_PULL
                    }
                },
                onTest1 = { send("1", "TEST S1") },
                onTest2 = { send("2", "TEST S2") },
                onPing = { send("?", "PING") }
            )
            Spacer(Modifier.height(12.dp))

            DigitalTwinCard(servo1 = servo1, servo2 = servo2)
            Spacer(Modifier.height(12.dp))

            CalibrationCard(
                servo1 = servo1,
                servo2 = servo2,
                onServo1 = {
                    servo1 = it
                    send("A${it.toInt().toString().padStart(3, '0')}\n", "Servo 1 ${it.toInt()}°")
                },
                onServo2 = {
                    servo2 = it
                    send("B${it.toInt().toString().padStart(3, '0')}\n", "Servo 2 ${it.toInt()}°")
                }
            )
            Spacer(Modifier.height(12.dp))

            HardwareInfoCard()
            Spacer(Modifier.height(12.dp))

            LogCard(logs = logs)
            Spacer(Modifier.height(18.dp))
        }
    }
}

@Composable
fun HeaderCard(status: String, connected: Boolean, deviceName: String) {
    ProCard {
        Row(verticalAlignment = Alignment.CenterVertically) {
            Canvas(Modifier.size(18.dp)) {
                drawCircle(if (connected) GREEN else RED)
            }
            Spacer(Modifier.width(10.dp))
            Column(Modifier.weight(1f)) {
                Text("SPIROB GRAPPLER", color = TEXT, fontSize = 22.sp, fontWeight = FontWeight.Bold)
                Text("Android Mission Control", color = MUTED, fontSize = 13.sp)
            }
            StatusPill(status)
        }
        Spacer(Modifier.height(10.dp))
        Text(
            text = "Uređaj: $deviceName",
            color = Color(0xFFCBD5E1),
            fontFamily = FontFamily.Monospace,
            fontSize = 12.sp,
            maxLines = 1,
            overflow = TextOverflow.Ellipsis
        )
    }
}

@Composable
fun StatusPill(status: String) {
    val color = when (status) {
        "ONLINE" -> GREEN
        "CONNECTING" -> AMBER
        else -> RED
    }
    Box(
        modifier = Modifier
            .background(color.copy(alpha = 0.18f), RoundedCornerShape(999.dp))
            .padding(horizontal = 12.dp, vertical = 7.dp)
    ) {
        Text(status, color = color, fontSize = 12.sp, fontWeight = FontWeight.Bold)
    }
}

@Composable
fun ConnectionCard(
    devices: List<BluetoothDevice>,
    selectedDevice: BluetoothDevice?,
    status: String,
    onRefresh: () -> Unit,
    onSelect: (BluetoothDevice) -> Unit,
    onConnect: () -> Unit,
    onDisconnect: () -> Unit
) {
    ProCard {
        SectionTitle("Konekcija")
        Spacer(Modifier.height(10.dp))

        Row {
            BlueButton("Osvježi", onRefresh, Modifier.weight(1f))
            Spacer(Modifier.width(8.dp))
            BlueButton("Poveži", onConnect, Modifier.weight(1f))
            Spacer(Modifier.width(8.dp))
            BlueButton("Prekini", onDisconnect, Modifier.weight(1f))
        }

        Spacer(Modifier.height(12.dp))

        Text("Upareni Bluetooth uređaji", color = MUTED, fontSize = 12.sp, fontWeight = FontWeight.Bold)
        Spacer(Modifier.height(6.dp))

        if (devices.isEmpty()) {
            Text("Nema pronađenih uparenih uređaja. Prvo upari HC-06 u Bluetooth postavkama.", color = MUTED, fontSize = 12.sp)
        } else {
            LazyColumn(
                modifier = Modifier
                    .fillMaxWidth()
                    .heightIn(max = 170.dp)
            ) {
                items(devices) { device ->
                    val selected = device.address == selectedDevice?.address
                    DeviceRow(device = device, selected = selected, onClick = { onSelect(device) })
                    Spacer(Modifier.height(6.dp))
                }
            }
        }

        Spacer(Modifier.height(8.dp))
        Text("Status: $status", color = if (status == "ONLINE") GREEN else if (status == "CONNECTING") AMBER else RED, fontFamily = FontFamily.Monospace, fontSize = 12.sp)
    }
}

@Composable
fun DigitalTwinCard(servo1: Float, servo2: Float) {
    val grip = gripIntensity(servo1, servo2)
    ProCard {
        Row(verticalAlignment = Alignment.CenterVertically) {
            SectionTitle("Digital Twin")
            Spacer(Modifier.weight(1f))
            Text("Grip intensity: $grip%", color = GREEN, fontFamily = FontFamily.Monospace, fontWeight = FontWeight.Bold, fontSize = 13.sp)
        }
        Spacer(Modifier.height(12.dp))
        Canvas(
            modifier = Modifier
                .fillMaxWidth()
                .height(260.dp)
                .background(Color(0xFF0B1120), RoundedCornerShape(18.dp))
                .padding(8.dp)
        ) {
            val cx = size.width / 2f
            val cy = size.height / 2f + 6f
            val ratio = grip / 100f
            val color = when {
                ratio < 0.45f -> Color(0xFF60A5FA)
                ratio < 0.80f -> Color(0xFFFACC15)
                else -> GREEN
            }

            drawCircle(Color(0xFF334155), radius = 64f, center = Offset(cx, cy), style = Stroke(width = 2f))

            for (i in 0 until 6) {
                val a = Math.toRadians((i * 60 - 90).toDouble())
                val curl = Math.toRadians((24 + 45 * ratio).toDouble())

                val x0 = cx + cos(a).toFloat() * 76f
                val y0 = cy + sin(a).toFloat() * 76f
                val x1 = x0 + cos(a + curl).toFloat() * 62f
                val y1 = y0 + sin(a + curl).toFloat() * 62f
                val x2 = x1 + cos(a + curl * 1.4).toFloat() * 40f
                val y2 = y1 + sin(a + curl * 1.4).toFloat() * 40f

                drawLine(color, Offset(x0, y0), Offset(x1, y1), strokeWidth = 6f, cap = StrokeCap.Round)
                drawLine(color, Offset(x1, y1), Offset(x2, y2), strokeWidth = 4.5f, cap = StrokeCap.Round)
                drawCircle(Color(0xFF94A3B8), radius = 5f, center = Offset(x0, y0))
                drawCircle(color, radius = 4f, center = Offset(x2, y2))
            }
        }
    }
}

fun gripIntensity(s1: Float, s2: Float): Int {
    val p1 = (SERVO1_HOME - s1) / (SERVO1_HOME - SERVO1_PULL)
    val p2 = (s2 - SERVO2_HOME) / (SERVO2_PULL - SERVO2_HOME)
    return (((p1.coerceIn(0f, 1f) + p2.coerceIn(0f, 1f)) / 2f) * 100f).toInt()
}

@Composable

    MaterialTheme(
        colorScheme = darkColorScheme(
            primary = BLUE,
            background = BG,
            surface = CARD,
            onPrimary = Color.White,
            onBackground = Color.White,
            onSurface = Color.White
        ),
        content = content
    )
}
