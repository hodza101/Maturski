class MainActivity : ComponentActivity() {
    private val sppUuid: UUID = UUID.fromString("00001101-0000-1000-8000-00805F9B34FB")

    private var socket: BluetoothSocket? = null
    private var output: OutputStream? = null
    private var input: InputStream? = null
    private var readerThread: Thread? = null
    private var onBluetoothMessage: ((String) -> Unit)? = null

    private val permissionLauncher = registerForActivityResult(
        ActivityResultContracts.RequestMultiplePermissions()
    ) {}

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        requestBluetoothPermissions()

        setContent {
            GrapplerTheme {
                GrapplerScreen(
                    getDevices = { getPairedDevices() },
                    connectDevice = { connectToDevice(it) },
                    sendCommand = { sendCommand(it) },
                    disconnect = { disconnect() },
                    setMessageListener = { listener -> onBluetoothMessage = listener }
                )
            }
        }
    }

    private fun requestBluetoothPermissions() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {
            permissionLauncher.launch(
                arrayOf(
                    Manifest.permission.BLUETOOTH_CONNECT,
                    Manifest.permission.BLUETOOTH_SCAN
                )
            )
        }
    }

    @SuppressLint("MissingPermission")
    private fun getPairedDevices(): List<BluetoothDevice> {
        val adapter = BluetoothAdapter.getDefaultAdapter() ?: return emptyList()

        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {
            val ok = ContextCompat.checkSelfPermission(
                this,
                Manifest.permission.BLUETOOTH_CONNECT
            ) == PackageManager.PERMISSION_GRANTED
            if (!ok) return emptyList()
        }

        return adapter.bondedDevices.toList().sortedWith(
            compareByDescending<BluetoothDevice> { (it.name ?: "").contains("HC", ignoreCase = true) }
                .thenBy { it.name ?: it.address }
        )
    }

    @SuppressLint("MissingPermission")
    private suspend fun connectToDevice(device: BluetoothDevice): Boolean = withContext(Dispatchers.IO) {
        try {
            disconnect()
            BluetoothAdapter.getDefaultAdapter()?.cancelDiscovery()

            val s = device.createRfcommSocketToServiceRecord(sppUuid)
            s.connect()

            socket = s
            output = s.outputStream
            input = s.inputStream
            startReaderThread()
            true
        } catch (_: Exception) {
            disconnect()
            false
        }
    }

    private fun sendCommand(cmd: String): Boolean {
        return try {
            val s = socket ?: return false
            val out = output ?: return false
            if (!s.isConnected) return false
            out.write(cmd.toByteArray(Charsets.US_ASCII))
            out.flush()
            true
        } catch (_: Exception) {
            false
        }
    }

    private fun startReaderThread() {
        readerThread = Thread {
            val buffer = StringBuilder()
            try {
                while (socket?.isConnected == true) {
                    val b = input?.read() ?: break
                    if (b == '\n'.code) {
                        val line = buffer.toString().trim()
                        buffer.clear()
                        if (line.isNotEmpty()) runOnUiThread { onBluetoothMessage?.invoke(line) }
                    } else if (b != '\r'.code) {
                        buffer.append(b.toChar())
                    }
                }
            } catch (_: Exception) {
            }
        }
        readerThread?.start()
    }

    private fun disconnect() {
        try { input?.close() } catch (_: Exception) {}
        try { output?.close() } catch (_: Exception) {}
        try { socket?.close() } catch (_: Exception) {}
        input = null
        output = null
        socket = null
        readerThread = null
    }

    override fun onDestroy() {
        disconnect()
        super.onDestroy()
    }
}

@Composable
fun GrapplerScreen(
    getDevices: () -> List<BluetoothDevice>,
    connectDevice: suspend (BluetoothDevice) -> Boolean,
    sendCommand: (String) -> Boolean,
    disconnect: () -> Unit,
    setMessageListener: (((String) -> Unit)?) -> Unit
) {
    val scope = rememberCoroutineScope()
    val scroll = rememberScrollState()

    var devices by remember { mutableStateOf<List<BluetoothDevice>>(emptyList()) }
    var selectedDevice by remember { mutableStateOf<BluetoothDevice?>(null) }
    var connected by remember { mutableStateOf(false) }
    var connecting by remember { mutableStateOf(false) }
    var status by remember { mutableStateOf("OFFLINE") }
    var servo1 by remember { mutableStateOf(SERVO1_HOME) }
    var servo2 by remember { mutableStateOf(SERVO2_HOME) }
    var logs by remember { mutableStateOf(listOf<String>()) }
    var lastCommand by remember { mutableStateOf("Nema") }

    fun now(): String = SimpleDateFormat("HH:mm:ss", Locale.getDefault()).format(Date())

    fun log(msg: String) {
        logs = (logs + "[${now()}] $msg").takeLast(80)
    }

    fun send(cmd: String, name: String): Boolean {
        if (!connected) {
            log("WARN: sistem nije povezan")
            return false
        }
        val ok = sendCommand(cmd)
        if (ok) {
            lastCommand = name
            log("TX: $name → '${cmd.trim()}'")
        } else {
            log("ERROR: komanda nije poslana")
        }
        return ok
    }

    LaunchedEffect(Unit) {
        devices = getDevices()
        log("Aplikacija pokrenuta")
        setMessageListener { line -> log("RX: $line") }
    }

    Surface(color = Bg, modifier = Modifier.fillMaxSize()) {
        Column(
            modifier = Modifier
                .fillMaxSize()
                .verticalScroll(scroll)
                .padding(14.dp)
        ) {
            Header(status = status, connected = connected, selectedName = selectedDevice?.name ?: "Nije izabran")
            Spacer(Modifier.height(12.dp))

            StatusOverview(
                connected = connected,
                connecting = connecting,
                deviceName = selectedDevice?.name ?: "—",
                lastCommand = lastCommand,
                grip = gripIntensity(servo1, servo2)
            )
            Spacer(Modifier.height(12.dp))

            CommandDeck(
                enabled = connected,
                onGrab = {
                    if (send("g", "Uhvati")) {
                        servo1 = SERVO1_PULL
                        servo2 = SERVO2_PULL
                    }
                },
                onStop = {
                    if (send("f", "E-STOP")) {
                        servo1 = SERVO1_HOME
                        servo2 = SERVO2_HOME
                    }
                },
                onHome = {
                    if (send("h", "Home")) {
                        servo1 = SERVO1_HOME
                        servo2 = SERVO2_HOME
                    }
                },
                onOpen = {
                    if (send("o", "Otvori")) {
                        servo1 = SERVO1_HOME
                        servo2 = SERVO2_HOME
                    }
                },
                onClose = {
                    if (send("c", "Zategni")) {
                        servo1 = SERVO1_PULL
                        servo2 = SERVO2_PULL
                    }
                },
                onTest1 = { send("1", "Test S1") },
                onTest2 = { send("2", "Test S2") },
                onPing = { send("?", "Ping") }
            )
            Spacer(Modifier.height(12.dp))

            DigitalTwin(servo1 = servo1, servo2 = servo2)
            Spacer(Modifier.height(12.dp))

            Calibration(
                enabled = connected,
                servo1 = servo1,
                servo2 = servo2,
                onServo1Change = { servo1 = it },
                onServo2Change = { servo2 = it },
                onServo1Send = { value -> send("A${value.toInt().toString().padStart(3, '0')}\n", "Servo 1 ${value.toInt()}°") },
                onServo2Send = { value -> send("B${value.toInt().toString().padStart(3, '0')}\n", "Servo 2 ${value.toInt()}°") }
            )
            Spacer(Modifier.height(12.dp))

            ConnectionPanel(
                devices = devices,
                selectedDevice = selectedDevice,
                connected = connected,
                connecting = connecting,
                onRefresh = {
                    devices = getDevices()
                    log("Lista uređaja osvježena")
                },
                onSelect = { selectedDevice = it },
                onConnect = {
                    val d = selectedDevice
                    if (d == null) {
                        log("WARN: nije izabran uređaj")
                    } else {
                        scope.launch {
                            connecting = true
                            status = "CONNECTING"
                            log("Povezivanje na ${d.name ?: d.address}...")
                            val ok = connectDevice(d)
                            connected = ok
                            connecting = false
                            status = if (ok) "ONLINE" else "OFFLINE"
                            if (ok) {
                                lastCommand = "Home"
                                log("OK: povezano na ${d.name ?: d.address}")
                                sendCommand("h")
                            } else {
                                log("ERROR: konekcija nije uspjela")
                            }
                        }
                    }
                },
                onDisconnect = {
                    disconnect()
                    connected = false
                    connecting = false
                    status = "OFFLINE"
                    log("Konekcija prekinuta")
                }
            )
            Spacer(Modifier.height(12.dp))

            HardwareCard()
            Spacer(Modifier.height(12.dp))

            LogPanel(logs = logs)
            Spacer(Modifier.height(18.dp))
        }
    }
}

@Composable
fun DigitalTwin(servo1: Float, servo2: Float) {
    val rawGrip = gripIntensity(servo1, servo2)
    val animatedGrip by animateFloatAsState(rawGrip.toFloat(), animationSpec = tween(550), label = "gripAnim")

    ProCard {
        Row(verticalAlignment = Alignment.CenterVertically) {
            SectionTitle("Digital Twin")
            Spacer(Modifier.weight(1f))
            Text("Grip ${animatedGrip.toInt()}%", color = Green, fontFamily = FontFamily.Monospace, fontWeight = FontWeight.Bold, fontSize = 13.sp)
        }
        Spacer(Modifier.height(12.dp))
        Canvas(
            modifier = Modifier
                .fillMaxWidth()
                .height(275.dp)
                .background(Panel2, RoundedCornerShape(20.dp))
                .padding(8.dp)
        ) {
            val cx = size.width / 2f
            val cy = size.height / 2f + 4f
            val ratio = (animatedGrip / 100f).coerceIn(0f, 1f)
            val limbColor = when {
                ratio < 0.45f -> Color(0xFF60A5FA)
                ratio < 0.80f -> Color(0xFFFACC15)
                else -> Green
            }

            drawCircle(Color(0xFF1E293B), radius = 74f, center = Offset(cx, cy), style = Stroke(width = 3f))
            drawCircle(Color(0xFF0F172A), radius = 60f, center = Offset(cx, cy))

            for (i in 0 until 6) {
                val a = Math.toRadians((i * 60 - 90).toDouble())
                val curl = Math.toRadians((18 + 54 * ratio).toDouble())
                val x0 = cx + cos(a).toFloat() * 78f
                val y0 = cy + sin(a).toFloat() * 78f
                val x1 = x0 + cos(a + curl).toFloat() * 64f
                val y1 = y0 + sin(a + curl).toFloat() * 64f
                val x2 = x1 + cos(a + curl * 1.45).toFloat() * 42f
                val y2 = y1 + sin(a + curl * 1.45).toFloat() * 42f

                drawLine(limbColor, Offset(x0, y0), Offset(x1, y1), strokeWidth = 7f, cap = StrokeCap.Round)
                drawLine(limbColor, Offset(x1, y1), Offset(x2, y2), strokeWidth = 5f, cap = StrokeCap.Round)
                drawCircle(Color(0xFFCBD5E1), radius = 5.5f, center = Offset(x0, y0))
                drawCircle(limbColor, radius = 4.5f, center = Offset(x2, y2))
            }
        }
    }
}

    onRefresh: () -> Unit,

    onConnect: () -> Unit,
    onDisconnect: () -> Unit
) {

        SectionTitle("Bluetooth konekcija")

            ActionButton("OSVJEŽI", Blue, true, onRefresh, Modifier.weight(1f))
            Spacer(Modifier.width(8.dp))

            Spacer(Modifier.width(8.dp))
            ActionButton("PREKINI", Blue, connected, onDisconnect, Modifier.weight(1f))

        Spacer(Modifier.height(12.dp))

            LazyColumn(modifier = Modifier.fillMaxWidth().heightIn(max = 190.dp)) {

                    DeviceRow(device, selected) { onSelect(device) }

                }
            }
        }
    }
}

@Composable
fun DeviceRow(device: BluetoothDevice, selected: Boolean, onClick: () -> Unit) {
    Box(
        modifier = Modifier
            .fillMaxWidth()
            .clip(RoundedCornerShape(16.dp))
            .background(if (selected) Blue.copy(alpha = 0.24f) else Panel2)
            .border(1.dp, if (selected) Blue else Border.copy(alpha = 0.45f), RoundedCornerShape(16.dp))
            .clickable { onClick() }
            .padding(13.dp)

    Column(

            .fillMaxWidth()
            .background(Panel, RoundedCornerShape(24.dp))

            .padding(16.dp),

    )
}

fun gripIntensity(s1: Float, s2: Float): Int {
    val p1 = (SERVO1_HOME - s1) / (SERVO1_HOME - SERVO1_PULL)
    val p2 = (s2 - SERVO2_HOME) / (SERVO2_PULL - SERVO2_HOME)
    return (((p1.coerceIn(0f, 1f) + p2.coerceIn(0f, 1f)) / 2f) * 100f).toInt()
}
